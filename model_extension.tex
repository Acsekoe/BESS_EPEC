\section{Model Direction}

\noindent The extension now returns to a spot-market EPEC. The central research object is competition for scarce accessible BESS power capacity at each node, followed by strategic spot-market operation through the lower-level DC-OPF market-clearing problem. Investor heterogeneity should no longer be limited to WACC. A later formulation should distinguish investors by operational constraints, for example whether the BESS is operated as a stand-alone asset or as part of a larger portfolio.

\subsection*{Lower-Level Problem --- Spot-Market Clearing}

\paragraph{Objective}
\begin{align*}
\min \quad
&\sum_{t,g} C_g P_{g,t}
\;+\;
\sum_{n,t} VOLL \, P^{shed}_{n,t}
\end{align*}
\nopagebreak[4]

\paragraph{Subject to}
\begin{align*}
&\sum_{g \in G_n} P_{g,t}
  + \sum_i \left(P^{discharge}_{i,n,t} - P^{charge}_{i,n,t}\right)
  + P^{shed}_{n,t}
  - D^{el}_{n,t}
  = NetInjection_{n,t},
  &&\forall n,t
  \quad (\lambda_{n,t})
\\
\intertext{This defines nodal net injections from generation, storage dispatch, load shedding, and demand.}
&\sum_n NetInjection_{n,t}=0,
  &&\forall t
  \quad (\lambda_{sys,t})
\\
\intertext{This enforces system-wide power balance.}
&0 \le P_{g,t} \le P^{max}_{g,t},
  &&\forall g,t
  \quad (\nu^-_{g,t},\nu^+_{g,t})
\\
\intertext{This keeps conventional and renewable generation within available capacity.}
&-Limit_l
  \le
  \sum_n PTDF_{l,n} NetInjection_{n,t}
  \le
  Limit_l,
  &&\forall l,t
  \quad (\mu^-_{l,t},\mu^+_{l,t})
\\
\intertext{This represents network feasibility through PTDF-based transmission limits.}
&0 \le P^{charge}_{i,n,t} \le X^{power}_{i,n},
  &&\forall i,n,t
  \quad (\rho^+_{i,n,t})
\\
&0 \le P^{discharge}_{i,n,t} \le X^{power}_{i,n},
  &&\forall i,n,t
  \quad (\sigma^+_{i,n,t})
\\
\intertext{These constraints limit storage charging and discharging by installed power capacity.}
&SOC_{i,n,t}
  =
  SOC_{i,n,t-1}
  + \eta P^{charge}_{i,n,t}
  - \frac{1}{\eta} P^{discharge}_{i,n,t},
  &&\forall i,n,t
  \quad (\gamma_{i,n,t})
\\
\intertext{This updates the BESS state of charge from spot-market operation only.}
&0 \le SOC_{i,n,\tau} \le X^{energy}_{i,n},
  &&\forall i,n,\tau
  \quad (\delta^-_{i,n,\tau},\delta^+_{i,n,\tau})
\\
&SOC_{i,n,0}=SOC_{i,n,24},
  &&\forall i,n
\\
\intertext{These constraints keep stored energy within installed capacity and impose daily SOC consistency.}
&0 \le P^{shed}_{n,t} \le D^{el}_{n,t},
  &&\forall n,t
  \quad (\xi^-_{n,t},\xi^+_{n,t})
\end{align*}

\bigskip
\noindent\textit{Exogenous parameters fixed from the upper level or input data:}
\begin{align*}
&X^{power}_{i,n},\; X^{energy}_{i,n},\; C_g,\; P^{max}_{g,t},\;
D^{el}_{n,t},\; Limit_l,\; PTDF_{l,n},\; \eta,\; VOLL.
\end{align*}

\subsection*{Upper-Level Investment Structure}

\noindent Each investor chooses nodal BESS power and energy capacity before the lower-level spot market clears. The shared nodal access constraint is the core competition mechanism:
\begin{align*}
&\sum_i X^{power}_{i,n} \le Limit^{node}_n,
  &&\forall n.
\end{align*}

\noindent A practical storage-sizing envelope should be retained:
\begin{align*}
&Ratio^{min} X^{power}_{i,n}
\le
X^{energy}_{i,n}
\le
Ratio^{max} X^{power}_{i,n},
&&\forall i,n.
\end{align*}

\noindent The upper-level objective should focus on expected or deterministic spot-market profit net of degradation and annualized CAPEX:
\begin{align*}
\max \; NPV_i
=&
\sum_{t,n}
\lambda^{LMP}_{n,t}
\left(P^{discharge}_{i,n,t}-P^{charge}_{i,n,t}\right)
- Cost^{degrad}_i
- CAPEX^{daily}_i.
\end{align*}

\noindent Investor heterogeneity can enter through WACC, CAPEX assumptions, degradation parameters, and operational restrictions such as stand-alone BESS operation versus portfolio-backed operation.
